    if(cache.get(obj)){
        return cache.get(obj)
    }

       let newObj = {}
       if(obj instanceof Array){

             return obj.call(this,...arguments)
         }

         newObj = new RegExp(obj.source,obj.flags)

           newObj = new Date(obj)

           newObj = {}
       }

}
