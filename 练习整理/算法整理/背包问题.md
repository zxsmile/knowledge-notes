function knapsack(weights,values,W){
    let n = weights.length
    let f = new Array(n)
    f[-1]=new Array(W+1).fill(0)

    for(var i=0;i<n;i++){
        f[i] = new Array(W+1).fill(0)
        for(var j=0;j<=W;j++){
            if(weights[i]>j){
               f[i][j] = f[i-1][j]
            }else{
                f[i][j] = Math.max(f[i-1][j],f[i-1][j-weights[i]]+values[i])
            }
           } 
       
    }
    return f[n-1][W]

}
